// Example:
Implement a last-in-first-out (LIFO) stack using only two queues. The implemented stack should support all the functions of a normal stack (push, top, pop, and empty).

Implement the MyStack class:

void push(int x) Pushes element x to the top of the stack.
int pop() Removes the element on the top of the stack and returns it.
int top() Returns the element on the top of the stack.
boolean empty() Returns true if the stack is empty, false otherwise.
Notes:

You must use only standard operations of a queue, which means that only push to back, peek/pop from front, size and is empty operations are valid.
Depending on your language, the queue may not be supported natively. You may simulate a queue using a list or deque (double-ended queue) as long as you use only a queue's standard operations.
 

// code:
class MyStack {
public:
queue<int>queueOrder,stackOrder;

    MyStack() {
        
    }
    
    void push(int x) {
       queueOrder.push(x);
    }
    
    int pop() {
        if (!queueOrder.empty()){
        while(queueOrder.size()!=1 ){
            stackOrder.push(queueOrder.front());
            queueOrder.pop();
        }
        int returnValue = queueOrder.front();
        queueOrder.pop();
         return returnValue;
        }
        while(stackOrder.size()!=1){
            queueOrder.push(stackOrder.front());
            stackOrder.pop();
        }
         int returnValue = stackOrder.front();
       stackOrder.pop();
         return returnValue;
       
        
    }
    
    int top() {
        if (!queueOrder.empty()){
            return queueOrder.back();
        }
            return stackOrder.back();
        
    }
    
    bool empty() {
        return queueOrder.empty() and stackOrder.empty();
    }
};
